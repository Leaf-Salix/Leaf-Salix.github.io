```cpp
int Lucas(int n,int m,int p)
{
    if(n<p&&m<p) return C(n,m,p);
    else return 1ll*Lucas(n/p,m/p,p)*C(n%p,m%p,p)%p;
}
```

char c;

while((c=getchar())!=EOF)

    while(c=='\n') c=getchar();
    if((c==' ')||(c==',')||(c=='.'))
    {
        query(s,len);
        len=0;
        if(c=='.') memset(b2,0,sizeof(b2));
        continue;/////
    }
    s[len++]=c;
}
